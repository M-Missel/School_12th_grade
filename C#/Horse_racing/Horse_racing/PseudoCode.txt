Operation waehlePferd():GZ
	Zufall zufall <- neuer Zufall
	RÜCKGABE zufall (0 BIS 10)
	
Operation endeRennen():BOOLEAN
	GZ Zaehler <- 0

	FÜR i <- 0 BIS LaengeArray Schritt 1
		WENN PferdeArray[i].Position == LaengeStrecke
			Zaehler <- Zaehler + 1

	WENN Zaehler == 10
		RÜCKGABE wahr
	SONST 
		RÜCKGABE falsch

Operation starteRennen()
	// Operation initPferde()

	GZ AuswahlPferd

	WÄHREND endeRennen == falsch
		AuswahlPferde <- waehlePferd()
		
		WENN PferdeArray[AuswahlPferde].Position < LaengeStrecke
			PferdeArray[AuswahlPferde].BewegePferd
			GUI.AktualisierePferde(AuswahlPferde, PferdeArray[AuswahlPferde].Nummer)
